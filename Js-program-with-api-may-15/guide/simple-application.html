Building a Simple API Application 🚀

Hey there, future developer! Today we're going to build a cool app that talks to an API. By the end of this guide, you'll have created your very own "Book Library Manager" that can:

  • Fetch books from an online database (GET)
  • Add new books to the collection (POST)
  • Update book information (PUT/PATCH)
  • Remove books (DELETE)
🧩 What We're Building

Imagine we're creating a digital bookshelf where you can see all your books, add new ones, edit information about them, and remove books you don't want anymore. The actual books are stored in an online database (the API), and our app is just the interface that helps you manage your collection.

// Book Library App JavaScript

5 Implementing the GET Request - Fetch Books

Let's add the function to fetch books from the API:

// Fetch all books from the API (GET)
async function fetchBooks() {
  try {
    // Show loading message
    loadingMessage.style.display = 'block';

    // Fetch data from API
    const response = await fetch(`${API_URL}/posts`);

    // Check if the request was successful
    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }

    // Parse the JSON response
    const books = await response.json();

    // Hide loading message
    loadingMessage.style.display = 'none';

    // Display the books
    displayBooks(books);

  } catch (error) {
    // Handle any errors
    console.error('Error fetching books:', error);
    loadingMessage.textContent = 'Error loading books. Please try again.';
    showStatusMessage('Error loading books from the server.', 'error');
  }
}

// Display books in the UI
function displayBooks(books) {
  // Clear the books container first
  booksContainer.innerHTML = '';

  // Check if we have any books
  if (books.length === 0) {
    booksContainer.innerHTML = '<p>No books found. Add some books!</p>';
    return;
  }

  // Loop through each book and create HTML for it
  books.forEach(book => {
    // Create a book card element
    const bookCard = document.createElement('div');
    bookCard.className = 'book-card';
    bookCard.dataset.id = book.id; // Store the book ID as a data attribute

    // In our test API, 'title' is the book title and 'body' is the description
    // We'll use 'userId' as the year published (just for this example)
    bookCard.innerHTML = `
      <div class="book-title">${book.title}</div>
      <div class="book-author">By: ${getAuthorNameFromBody(book.body)}</div>
      <div class="book-year">Published: ${book.userId + 2000}</div>
      <div class="book-actions">
        <button class="btn edit-btn" data-id="${book.id}">Edit</button>
        <button class="btn delete-btn" data-id="${book.id}">Delete</button>
      </div>
    `;

    // Add event listeners to the buttons
    const editBtn = bookCard.querySelector('.edit-btn');
    const deleteBtn = bookCard.querySelector('.delete-btn');

    editBtn.addEventListener('click', () => openEditModal(book));
    deleteBtn.addEventListener('click', () => deleteBook(book.id));

    // Add the book card to the container
    booksContainer.appendChild(bookCard);
  });
}

// Helper function to extract an author name from the body text
// In a real app, you'd have proper author data
function getAuthorNameFromBody(body) {
  // Just grab the first word as a fake author name
  const firstWords = body.split(' ').slice(0, 2).join(' ');
  return firstWords.charAt(0).toUpperCase() + firstWords.slice(1);
}

6 Implementing the POST Request - Add a New Book

Now let's implement the functionality to add a new book to the library:

// Handle adding a new book (POST)
async function handleAddBook(e) {
  e.preventDefault(); // Prevent the form from submitting normally

  // Get the values from the form
  const title = document.getElementById('title').value.trim();
  const author = document.getElementById('author').value.trim();
  const year = parseInt(document.getElementById('year').value);

  // Basic validation
  if (!title || !author || !year) {
    showStatusMessage('Please fill in all fields.', 'error');
    return;
  }

  // For our JSONPlaceholder API, we need to format it like a 'post'
  // In a real book API, the structure would match your book data model
  const newBook = {
    title: title,
    body: `Written by ${author}. ${title} is a great book.`, // We'll store author in the body
    userId: year - 2000 // We'll pretend userId is related to year
  };

  try {
    // Send POST request to create a new book
    const response = await fetch(`${API_URL}/posts`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(newBook)
    });

    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }

    // Parse the response data
    const createdBook = await response.json();

    // Show success message
    showStatusMessage(`Book "${title}" added successfully!`, 'success');

    // Reset the form
    addBookForm.reset();

    // In a real app, we might refresh the book list here
    // For now, let's just add the new book to our display
    const updatedBooks = await fetch(`${API_URL}/posts`).then(res => res.json());
    displayBooks(updatedBooks);

  } catch (error) {
    console.error('Error adding book:', error);
    showStatusMessage('Error adding book. Please try again.', 'error');
  }
}

7 Implementing the PUT/PATCH Request - Edit a Book

Let's add the functionality to edit an existing book:

// Open the edit modal with book data
function openEditModal(book) {
  // Set the book ID in the hidden field
  document.getElementById('edit-book-id').value = book.id;

  // Fill in the form with the book's current data
  document.getElementById('edit-title').value = book.title;

  // Extract author from the body text (in a real app, you'd have proper author data)
  document.getElementById('edit-author').value = getAuthorNameFromBody(book.body);

  // Calculate year from userId (just for our example)
  document.getElementById('edit-year').value = book.userId + 2000;

  // Show the modal
  editModal.style.display = 'block';
}

// Handle the edit book form submission (PUT)
async function handleEditBook(e) {
  e.preventDefault(); // Prevent the form from submitting normally

  // Get the values from the form
  const bookId = document.getElementById('edit-book-id').value;
  const title = document.getElementById('edit-title').value.trim();
  const author = document.getElementById('edit-author').value.trim();
  const year = parseInt(document.getElementById('edit-year').value);

  // Basic validation
  if (!title || !author || !year) {
    showStatusMessage('Please fill in all fields.', 'error');
    return;
  }

  // Prepare the updated book data
  const updatedBook = {
    id: bookId, // Include the ID for updating
    title: title,
    body: `Written by ${author}. ${title} is a great book.`,
    userId: year - 2000
  };

  try {
    // Send PUT request to update the book
    // We could also use PATCH if we only wanted to update specific fields
    const response = await fetch(`${API_URL}/posts/${bookId}`, {
      method: 'PUT',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(updatedBook)
    });

    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }

    // Parse the response data
    const updatedBookData = await response.json();

    // Close the modal
    editModal.style.display = 'none';

    // Show success message
    showStatusMessage(`Book "${title}" updated successfully!`, 'success');

    // Refresh the book list
    fetchBooks();

  } catch (error) {
    console.error('Error updating book:', error);
    showStatusMessage('Error updating book. Please try again.', 'error');
  }
}

// Example of a PATCH request (only updating some fields)
// This function shows how to use PATCH instead of PUT
async function updateBookPartially(bookId, updateData) {
  try {
    const response = await fetch(`${API_URL}/posts/${bookId}`, {
      method: 'PATCH', // Use PATCH for partial updates
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(updateData) // Only send the fields that changed
    });

    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }

    return await response.json();

  } catch (error) {
    console.error('Error with PATCH update:', error);
    throw error;
  }
}

8 Implementing the DELETE Request - Remove a Book

Finally, let's add the functionality to delete books:

// Delete a book (DELETE)
async function deleteBook(bookId) {
  // Ask for confirmation
  if (!confirm('Are you sure you want to delete this book?')) {
    return; // User cancelled
  }

  try {
    // Send DELETE request
    const response = await fetch(`${API_URL}/posts/${bookId}`, {
      method: 'DELETE'
    });

    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }

    // Show success message
    showStatusMessage('Book deleted successfully!', 'success');

    // Remove the book from the UI
    const bookCard = document.querySelector(`.book-card[data-id="${bookId}"]`);
    if (bookCard) {
      bookCard.remove();
    }

    // If no books left, show message
    if (booksContainer.children.length === 0) {
      booksContainer.innerHTML = '<p>No books found. Add some books!</p>';
    }

  } catch (error) {
    console.error('Error deleting book:', error);
    showStatusMessage('Error deleting book. Please try again.', 'error');
  }
}

11 Debugging Common Issues

If something isn't working, here are some common issues to check:

  • 404 errors: Make sure your API URLs are correct
  • CORS errors: This happens when an API doesn't allow requests from your domain. JSONPlaceholder should work fine, but other APIs might need special permission.
  • Data not displaying correctly: Use console.log() to check what data you're getting from the API
  • JavaScript errors: Check the Console tab in Developer Tools for any error messages
