üîó React + Tailwind Integration

Component Styling Patterns

// Card Component Example
function Card({ title, children, variant = "default" }) {
  const baseClasses = "rounded-lg shadow-md p-6";
  const variants = {
    default: "bg-white border border-gray-200",
    primary: "bg-blue-50 border border-blue-200",
    success: "bg-green-50 border border-green-200"
  };
  
  return (
    <div className={`${baseClasses} ${variants[variant]}`}>
      <h3 className="text-lg font-semibold mb-4">{title}</h3>
      {children}
    </div>
  );
}

Conditional Styling

function Button({ isActive, children }) {
  return (
    <button 
      className={`px-4 py-2 rounded font-medium transition-colors ${
        isActive 
          ? 'bg-blue-500 text-white hover:bg-blue-600' 
          : 'bg-gray-200 text-gray-700 hover:bg-gray-300'
      }`}
    >
      {children}
    </button>
  );
}

Form Components

function InputField({ label, type = "text", ...props }) {
  return (
    <div className="mb-4">
      <label className="block text-sm font-medium text-gray-700 mb-2">
        {label}
      </label>
      <input 
        type={type}
        className="w-full px-3 py-2 border border-gray-300 rounded-md 
                   focus:outline-none focus:ring-2 focus:ring-blue-500"
        {...props}
      />
    </div>
  );
}

Best Practices

  • Use template literals for conditional classes
  • Create reusable component variants
  • Leverage Tailwind's component extraction for repeated patterns
  • Use CSS variables for theme customization
